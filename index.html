<script>
let dragged = null;

// 4. Gerador de HTML e CSS (Baseado no seu código)
let baseCSS = `
.nav-menu ul { list-style: none; padding: 0; margin: 0; display: flex; font-family: sans-serif; }
.nav-menu li { position: relative; }
.nav-menu a { display: block; background: #2563eb; padding: 15px 25px; color: white; text-decoration: none; border-right: 1px solid transparent; }
.nav-menu a:hover { background: #1d4ed8; opacity: .8; }
.nav-menu ul ul { display: none; position: absolute; top: 100%; left: 0; flex-direction: column; min-width: 200px; z-index: 100; }
.nav-menu li:hover > ul { display: flex; }
.nav-menu ul ul ul { top: 0; left: 100%; background: #1e40af; }
.nav-menu ul ul a { border-right: none; border-bottom: 1px solid #3b82f6; }
`.trim();

let novaCor;

document.querySelector('#color').addEventListener('input', (e) => {
novaCor = e.target.value;
// Em vez de replace fixo, atualizamos a variável no CSS dinâmico
// ou reconstruímos a baseCSS usando um marcador.
updateCSS(novaCor);
});

// Melhore sua função de atualização para aceitar a cor
function updateCSS(cor) {
if(!novaCor) return;
// Sobrescreve apenas a cor principal mantendo o restante
const cssAtualizado = baseCSS.replaceAll('#2563eb', cor)
.replaceAll('#1d4ed8', cor) // tom mais escuro opcional
.replaceAll('#3b82f6', cor); // bordas
document.getElementById('css-code').innerText = cssAtualizado;
document.getElementById('dynamic-menu-css').innerHTML = cssAtualizado;
}

// 1. Função centralizada para criar o elemento LI (Evita repetição)
function createMenuItemElement(name, url = "#") {
const li = document.createElement('li');
li.className = 'menu-item';
li.draggable = true;
li.dataset.label = name;
li.dataset.url = url;

li.innerHTML = `
<span>${name}</span>
<button class="delete-btn" title="Remover">X</button>
`;

// Evento de deletar
li.querySelector('.delete-btn').addEventListener('click', (e) => {
e.stopPropagation();
if (confirm(`Remover "${name}" e seus submenus?`)) {
li.remove();
update();
}
});

initItem(li);
return li;
}

// 2. Inicializa os eventos de Drag & Drop
function initItem(el) {
el.addEventListener('dragstart', (e) => {
dragged = el;
e.stopPropagation();
el.style.opacity = "0.4";
});

el.addEventListener('dragend', () => {
el.style.opacity = "1";
});

el.addEventListener('dragover', (e) => {
e.preventDefault();
e.stopPropagation();
el.classList.add('drag-over');
});

el.addEventListener('dragleave', () => el.classList.remove('drag-over'));

el.addEventListener('drop', (e) => {
e.preventDefault();
e.stopPropagation();
el.classList.remove('drag-over');
if (dragged === el) return;
// Impede de arrastar um pai para dentro de um próprio filho (evita erro infinito)
if (el.contains(dragged)) return;

let sub = el.querySelector('ul');
if (!sub) {
sub = document.createElement('ul');
sub.className = 'drag-list';
el.appendChild(sub);
}
sub.appendChild(dragged);
update();
});
}

// 3. Adiciona novo item via Input
function addItem() {
const nameInput = document.getElementById('newItemName');
const urlInput = document.getElementById('newItemUrl'); // Adicione este input no HTML se desejar
const name = nameInput.value || 'Novo Item';
const url = (urlInput && urlInput.value) ? urlInput.value : '#';

const newLi = createMenuItemElement(name, url);
document.getElementById('root-list').appendChild(newLi);
nameInput.value = '';
urlInput.value = '';
update();
}


function getHTML(ul) {
let html = '<ul>';
Array.from(ul.querySelectorAll(':scope > li')).forEach((li) => {
// Link limpo para o código final
html += `\n <li><a href="${li.dataset.url}">${li.dataset.label}</a>`;
const sub = li.querySelector('ul');
if (sub) {
html += getHTML(sub).split('\n').map(line => ' ' + line).join('\n');
}
html += "</li>";
});
return html + '\n</ul>';
}

function update() {
const root = document.getElementById('root-list');
const menuHTML = getHTML(root);
const finalHTML = `<nav class="nav-menu">\n${menuHTML}\n</nav>`;

document.getElementById('html-code').innerText = finalHTML;
document.getElementById('css-code').innerText = baseCSS;

// Preview
const previewArea = document.getElementById('preview-area');
previewArea.innerHTML = finalHTML;
// Desabilitar links no preview para não navegar ao testar o menu
previewArea.querySelectorAll('a').forEach(a => {
a.addEventListener('click', e => e.preventDefault());
});

document.getElementById('dynamic-menu-css').innerHTML = baseCSS;
updateCSS(novaCor);
}

// Inicialização
document.querySelectorAll('.menu-item').forEach(li => {
// Se o li já existe no HTML, vamos reconstruir para ter a lixeira
const name = li.dataset.label || li.innerText.trim();
const url = li.dataset.url || "#";
const newLi = createMenuItemElement(name, url);
li.replaceWith(newLi);
});

update();
</script>
